 *
 * Battery template override — adds PIP appliance parts to all vehicles.
 * PZ template overrides REPLACE the entire template, so we must
 * re-declare the vanilla Battery part to preserve vehicle functionality.
 * Towbar parts are NOT included — the Towbar mod handles its own override.
 */

module Base
{
    template vehicle Battery
    {
        /* -------------------------------------------------------
         * Vanilla: Car Battery (must be re-declared — template
         * overrides replace, not merge)
         * ------------------------------------------------------- */
        part Battery
        {
            mechanicArea = Engine,
            area = Engine,
            itemType = Base.CarBattery,
            mechanicRequireKey = true,
            category = engine,
            durability = 2,

            table install
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 100,
                test = Vehicles.InstallTest.Default,
                door = EngineDoor,
            }

            table uninstall
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 100,
                test = Vehicles.UninstallTest.Battery,
            }

            lua
            {
                create = Vehicles.Create.Battery,
                update = Vehicles.Update.Battery,
            }
        }

        /* -------------------------------------------------------
         * PIP: Portable Lab Fridge
         * ------------------------------------------------------- */
        part PIPLabFridge
        {
            area = TrunkBed,
            itemType = PhobosIndustrialPathology.PortableLabFridge,
            mechanicRequireKey = true,
            category = Other,

            container
            {
                capacity = 15,
                test = PIP_Vehicle.ContainerAccess.Fridge,
            }

            table install
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 200,
                skills = Electricity:2,
                test = Vehicles.InstallTest.Default,
            }

            table uninstall
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 200,
                requireEmpty = true,
                test = Vehicles.UninstallTest.Default,
            }

            lua
            {
                create = PIP_Vehicle.Create.Appliance,
                init = PIP_Vehicle.Init.Fridge,
                update = PIP_Vehicle.Update.Fridge,
            }
        }

        /* -------------------------------------------------------
         * PIP: Portable Lab Microwave
         * ------------------------------------------------------- */
        part PIPLabMicrowave
        {
            area = TrunkBed,
            itemType = PhobosIndustrialPathology.PortableLabMicrowave,
            mechanicRequireKey = true,
            category = Other,

            container
            {
                capacity = 10,
                test = PIP_Vehicle.ContainerAccess.Microwave,
            }

            table install
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 200,
                skills = Electricity:3,
                test = Vehicles.InstallTest.Default,
            }

            table uninstall
            {
                items
                {
                    1
                    {
                        tags = base:screwdriver,
                        count = 1,
                        keep = true,
                        equip = primary,
                    }
                }
                time = 200,
                requireEmpty = true,
                test = Vehicles.UninstallTest.Default,
            }

            lua
            {
                create = PIP_Vehicle.Create.Appliance,
                init = PIP_Vehicle.Init.Microwave,
                update = PIP_Vehicle.Update.Microwave,
            }
        }
    }
}
